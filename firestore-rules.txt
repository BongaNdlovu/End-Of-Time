rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Leaderboard collection rules
    match /leaderboard/{userId} {
      // Anyone can read leaderboard entries (for displaying the leaderboard)
      allow read: if true;

      // Enforce best-only writes on the server:
      // - Create: allowed for owner
      // - Update: allowed only if new score is higher, or equal score with lower (faster) time
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId
                    && (
                      request.resource.data.score > resource.data.score ||
                      (request.resource.data.score == resource.data.score && request.resource.data.time < resource.data.time)
                    );
    }
    
    // Test collection for validation (optional)
    match /test/{document} {
      allow read, write: if request.auth != null;
    }
  }
}
